$LOOP$ =
 user_clip action_if_live_onpress SEL/CLIP LOOP :
 user_clip action_if_live_onrelease NONE

$LOOP_DOUBLE$ =
 SEL/CLIP LOOP *2;
 MSG "Doubled loop size"

$LOOP_HALVE$ =
 SEL/CLIP LOOP *.5;
 MSG "Halved loop size"

$LOOP_LEFT$ =
 SEL/CLIP START <1B;
 MSG "Moved clip start back 1 bar"

$LOOP_NUDGE_LEFT$ =
 SEL/JUMP <;
 MSG "Jumped clip back"

$LOOP_NUDGE_RIGHT$ =
 SEL/JUMP >;
 MSG "Jumped clip forward"

$LOOP_RECORD$ =
 MSG "Recording new clip";
 user_clip select_traktor_deck;
 5-8/ARM OFF;
 SEL/ARM ON;
 SEL/SEL EMPTY;
 SRECFIX 4;
 WAITS 5B;
 SEL/ARM OFF;
 user_clip deselect_traktor_deck;
 MSG "Recorded new clip"

$LOOP_RIGHT$ =
 SEL/CLIP START >1B;
 MSG "Moved clip start forward 1 bar"
